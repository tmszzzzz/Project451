# 人物定义
define 1 'Steve'
define 2 'Alex'
define 3 'Me'

# 对话
say 1 '这是一个很长的对话文本，可以换行。\n\
这个文本包含了多个句子，继续描述剧情。'
say 2 '这是另一个角色的长文本对话，\n\
同样可以包含多行，并且可以自由描述对话内容。'
Isay 3 '我自己的长文本对话，\n\
继续讲述剧情细节。'

# 选择项
selectItem '选择A' flag1
selectItem '选择B' flag2
selectItem '选择C' flag3
select 

# 分支处理
flag flag1
Isay 3 '我选择了A，接下来发生了什么...'
goto end

flag flag2
Isay 3 '我选择了B，事情变得复杂了...'
goto end

flag flag3
Isay 3 '我选择了C，结果出乎意料...'
goto end

flag end

# 剧情退出
exit
